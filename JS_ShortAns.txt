Write a JavaScript function, makeStars(num), that returns a string of [num] star characters ('*'). For example, calling makeStars(5) should return '*****'. Do not use the 'repeat' string method.

function makeStars(num) {
    let stars = '';
    for (let i = 0; i < num; i++) {
         stars += "*";
    }
    return stars;
}
